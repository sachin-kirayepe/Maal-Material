
// ==========================================
// EQUIPMENT RENTAL & FLEET MANAGEMENT
// ==========================================

model EquipmentAsset {
  id               String   @id @default(uuid())
  tenantId         String
  ownerId          String
  name             String
  category         String   // HEAVY_MACHINERY, POWER_EQUIPMENT, TOOLS
  model            String
  vinOrSerial      String?  @unique
  status           String   @default("AVAILABLE") // AVAILABLE, RENTED, MAINTENANCE, INACTIVE
  location         String?
  yearOfMake       Int?
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt
  
  pricing          RentalPricing?
  availability     EquipmentAvailability[]
  bookings         RentalBooking[]
  maintenance      MaintenanceSchedule[]
  deployments      EquipmentDeployment[]
}

model RentalPricing {
  id               String   @id @default(uuid())
  equipmentId      String   @unique
  hourlyRate       Float?
  dailyRate        Float?
  weeklyRate       Float?
  monthlyRate      Float?
  operatorCharge   Float?
  dieselCharge     Float?
  depositAmount    Float?
  updatedAt        DateTime @updatedAt
  
  equipment        EquipmentAsset @relation(fields: [equipmentId], references: [id], onDelete: Cascade)
}

model EquipmentAvailability {
  id               String   @id @default(uuid())
  tenantId         String
  equipmentId      String
  startDate        DateTime
  endDate          DateTime
  status           String   @default("BLOCKED") // BLOCKED, BUFFER
  reason           String?  // BOOKED, MAINTENANCE
  createdAt        DateTime @default(now())
  
  equipment        EquipmentAsset @relation(fields: [equipmentId], references: [id], onDelete: Cascade)
  
  @@index([equipmentId, startDate, endDate])
}

model RentalBooking {
  id               String   @id @default(uuid())
  tenantId         String
  equipmentId      String
  contractorId     String
  startDate        DateTime
  endDate          DateTime
  totalAmount      Float
  status           String   @default("REQUESTED") // REQUESTED, APPROVED, ACTIVE, COMPLETED, CANCELLED
  operatorAssigned Boolean  @default(false)
  deliveryAddress  String?
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt
  
  equipment        EquipmentAsset @relation(fields: [equipmentId], references: [id])
}

model FleetOperation {
  id               String   @id @default(uuid())
  tenantId         String
  ownerId          String
  totalAssets      Int      @default(0)
  activeRentals    Int      @default(0)
  inMaintenance    Int      @default(0)
  utilizationRate  Float    @default(0.0)
  updatedAt        DateTime @updatedAt
  
  @@unique([tenantId, ownerId])
}

model MaintenanceSchedule {
  id               String   @id @default(uuid())
  tenantId         String
  equipmentId      String
  maintenanceType  String   // PREVENTATIVE, REPAIR, INSPECTION
  scheduledDate    DateTime
  completedDate    DateTime?
  cost             Float?
  notes            String?
  status           String   @default("SCHEDULED") // SCHEDULED, IN_PROGRESS, COMPLETED
  createdAt        DateTime @default(now())
  
  equipment        EquipmentAsset @relation(fields: [equipmentId], references: [id], onDelete: Cascade)
}

model EquipmentDeployment {
  id               String   @id @default(uuid())
  tenantId         String
  equipmentId      String
  bookingId        String?
  transporterId    String?
  origin           String
  destination      String
  dispatchDate     DateTime?
  arrivalDate      DateTime?
  status           String   @default("PENDING") // PENDING, IN_TRANSIT, DELIVERED, RETURNED
  createdAt        DateTime @default(now())
  
  equipment        EquipmentAsset @relation(fields: [equipmentId], references: [id], onDelete: Cascade)
}

model RentalRFQ {
  id               String   @id @default(uuid())
  tenantId         String
  contractorId     String
  equipmentType    String
  quantity         Int      @default(1)
  requiredFrom     DateTime
  requiredUntil    DateTime
  location         String
  status           String   @default("OPEN") // OPEN, NEGOTIATING, FULFILLED, CANCELLED
  bidsReceived     Int      @default(0)
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt
}

model OperatorAssignment {
  id               String   @id @default(uuid())
  tenantId         String
  operatorId       String
  bookingId        String
  equipmentId      String
  assignedDate     DateTime @default(now())
  status           String   @default("ASSIGNED") // ASSIGNED, COMPLETED
}

model FleetAnalytics {
  id               String   @id @default(uuid())
  tenantId         String
  fleetOwnerId     String
  metricKey        String   // REVENUE, UTILIZATION, MAINTENANCE_COST
  metricValue      Float
  recordedAt       DateTime @default(now())
  
  @@index([tenantId, fleetOwnerId])
}
